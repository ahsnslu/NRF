#include <SPI.h>
#include <nRF24L01.h>
#include <RF24.h>
#include <Servo.h>

// NRF24L01 radyo modülü için tanımlamalar
RF24 radio(9, 10); // CE, CSN
const byte address[6] = "00001";

// Servo için tanımlamalar
Servo myServo;
int servoPin = 3;

// Gelen verilerin yapısı
struct DataPackage {
  bool buttonPressed;
};

DataPackage data;

void setup() {
  Serial.begin(9600);

  // Servo pinini ayarla
  myServo.attach(servoPin);

  // Radyo modülünü başlat
  radio.begin();
  radio.setDataRate(RF24_2MBPS); // Veri hızını 2 Mbps olarak ayarla
  radio.openReadingPipe(0, address);
  radio.setPALevel(RF24_PA_MIN);
  radio.startListening();
}

void loop() {
  if (radio.available()) {
    radio.read(&data, sizeof(data));

    // Seri port üzerinden veri gönder
    Serial.print("Button Pressed: ");
    Serial.println(data.buttonPressed);

    // Butona basıldığında servo motoru 45 dereceye döndürüp geri getir
    if (data.buttonPressed) {
      myServo.write(0); // Servo motoru 45 dereceye döndür
      delay(1000); // 1 saniye bekle
      myServo.write(80); // Servo motoru başlangıç pozisyonuna döndür

      // Tekrar buttonPressed'i false yaparak iki kez çalışmasını önle
      data.buttonPressed = false;
    }
  }

  delay(50); // Kısa bir gecikme
}

#include <SPI.h>
#include <nRF24L01.h>
#include <RF24.h>

// NRF24L01 radyo modülü için tanımlamalar
RF24 radio(9, 10); // CE, CSN
const byte address[6] = "00001";

// Joystick için tanımlamalar
int buttonPin = 7; // Buton için dijital pin (Joystick modülündeki "SW" pini)
int ledPin = 8; // LED için pin (atış durumunu göstermek için)

struct DataPackage {
  bool buttonPressed;
};

DataPackage data;

void setup() {
  Serial.begin(9600);

  // Joystick buton pinini giriş olarak ayarla
  pinMode(buttonPin, INPUT_PULLUP);
  pinMode(ledPin, OUTPUT);

  // Radyo modülünü başlat
  radio.begin();
  radio.setDataRate(RF24_2MBPS); // Veri hızını 2 Mbps olarak ayarla
  radio.openWritingPipe(address);
  radio.setPALevel(RF24_PA_MIN);
}

void loop() {
  // Joystick buttonPressed değerini oku
  data.buttonPressed = digitalRead(buttonPin) == LOW; // Buton basılıysa LOW (0)

  // Veriyi radyo ile gönder
  radio.stopListening();
  radio.write(&data, sizeof(data));
  radio.startListening();

  // Buton basıldığında LED'i yak
  if (data.buttonPressed) {
    digitalWrite(ledPin, HIGH); // LED'i yak
    delay(1000); // 1 saniye bekle
    digitalWrite(ledPin, LOW); // LED'i söndür
  }

  // Seri port üzerinden veri gönder
  Serial.print("Button Pressed: ");
  Serial.println(data.buttonPressed);

  delay(50); // Kısa bir gecikme
}
